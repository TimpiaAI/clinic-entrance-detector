```typescript
/**
 * Show crash alert overlay when detector unexpectedly stops.
 * Displays a Romanian message with a Restart button.
 */
export function showCrashAlert(onRestart: () => void): void {
  const alert = document.getElementById('crash-alert');
  const text = document.getElementById('crash-alert-text');
  const btn = document.getElementById('crash-restart-btn');

  if (text) text.textContent = RO.CRASH_ALERT;
  if (btn) {
    btn.textContent = RO.CRASH_RESTART;
    // Remove old listeners by cloning
    const newBtn = btn.cloneNode(true) as HTMLButtonElement;
    btn.parentNode?.replaceChild(newBtn, btn);
    newBtn.addEventListener('click', onRestart, { once: true });
  }
  if (alert) alert.classList.add('visible');
}

/**
 * Hide the crash alert overlay.
 */
export function hideCrashAlert(): void {
  const alert = document.getElementById('crash-alert');
  if (alert) alert.classList.remove('visible');
}
```

**stopSystem()** (CTRL-01):
1. `updateSystemButton(true, true)` -- show "Se opreste..."
2. `stopWorkflow()` -- abort workflow, hide video, clear data
3. `stopHealthMonitor()`
4. `await apiStopDetector()`
5. `await apiWakeLockRelease()`
6. `hideCrashAlert()`
7. `updateSystemButton(false)` -- show "Start"
8. Set `systemRunning = false`

**emergencyStop()** (CTRL-02):
Same as stopSystem but no transitioning UI -- immediate. Call `stopWorkflow()` first (aborts recording), then `apiStopDetector()`, `apiWakeLockRelease()`, `hideTranscriptionPanel()`, `hideCrashAlert()`.

**toggleSystem():**
If running, call stopSystem(). If not, call startSystem(). This is what F2 and the button wire to.

**autoStart()** (CTRL-03):
On page load, auto-start the detector and wake-lock (no user gesture needed for these). Do NOT initialize audio (requires gesture). Do NOT start workflow yet -- wait for F2 press.
Check for `?no-autostart` query param for dev mode skip.

```typescript
export async function autoStart(): Promise<void> {
  if (new URLSearchParams(window.location.search).has('no-autostart')) return;
```
