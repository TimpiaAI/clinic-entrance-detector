```python
async def receive_trigger(request: Request) -> JSONResponse:
```

```typescript
/**
 * Show all captured patient data for final confirmation before submission.
 * Reuses #transcription-panel with a summary layout showing name, question, CNP, email.
 */
export function showConfirmationSummary(
  data: PatientData,
  onConfirm: () => void,
  onCancel: () => void,
): void {
  const panel = document.getElementById('transcription-panel');
  const indicator = document.getElementById('recording-indicator');
  const statusText = document.getElementById('status-text');
  const result = document.getElementById('transcription-result');
  const actions = document.getElementById('transcription-actions');
  const resultText = document.getElementById('result-text');
  const resultCnp = document.getElementById('result-cnp');
  const resultEmail = document.getElementById('result-email');

  if (panel) panel.classList.add('visible');
  if (indicator) indicator.classList.remove('active');
  if (statusText) statusText.textContent = RO.WORKFLOW_CONFIRM_ALL;

  // Build summary in result-text area
  const lines: string[] = [];
  if (data.name) lines.push(`${RO.WORKFLOW_NAME_LABEL}: ${data.name}`);
  if (data.question) lines.push(`${RO.WORKFLOW_QUESTION_LABEL}: ${data.question}`);
  if (resultText) resultText.textContent = lines.join('\n');

  if (resultCnp) {
    resultCnp.textContent = data.cnp ? `${RO.CNP_LABEL}: ${data.cnp}` : '';
    resultCnp.style.display = data.cnp ? 'block' : 'none';
  }
  if (resultEmail) {
    resultEmail.textContent = data.email ? `${RO.EMAIL_LABEL}: ${data.email}` : '';
    resultEmail.style.display = data.email ? 'block' : 'none';
  }
  if (result) result.classList.add('visible');

  if (actions) {
    actions.innerHTML = '';
    const confirmBtn = document.createElement('button');
    confirmBtn.className = 'btn-confirm';
    confirmBtn.textContent = RO.WORKFLOW_CONFIRM_SEND;
    confirmBtn.addEventListener('click', onConfirm, { once: true });

    const cancelBtn = document.createElement('button');
    cancelBtn.className = 'btn-retry';
    cancelBtn.textContent = RO.WORKFLOW_CONFIRM_CANCEL;
    cancelBtn.addEventListener('click', onCancel, { once: true });

    actions.appendChild(confirmBtn);
    actions.appendChild(cancelBtn);
    actions.classList.add('visible');
  }
}
```

**6. Create frontend/src/workflow.ts** -- The state machine:

This is the core module. Build it following the RESEARCH.md patterns exactly. Key design:

- Module-level state: `currentState: WorkflowState = 'stopped'`, `patientData: PatientData`, `stateTimeout`, `activeAbort` for recording cancellation.
- `transition(newState)`: clears timeout, sets new state, starts state-specific timeout, calls `executeStateEntry(newState)`.
- `executeStateEntry()`: big switch on state that orchestrates video, recording, and UI.
- `handleTimeout()`: aborts active recording if any, clears patient data, transitions to idle.
- State timeouts from RESEARCH.md: 60s for video states, 15s for recording, 30s for show/confirm.

Exact workflow sequence from controller.py:
```
idle (video1 loops)
  -> person_entered event
  -> greeting (play video2 once)
  -> video2 ends -> ask_name (play video3 once)
  -> video3 ends -> recording_name (video1 loops, record 10s, show panel)
  -> confirm/retry -> show_name result
  -> confirmed -> ask_question (play video6 once)
  -> video6 ends -> recording_question (video1 loops, record 10s)
  -> confirmed -> ask_cnp (play video7 once, prompt="1 2 3...")
  -> video7 ends -> recording_cnp (video1 loops, record 10s)
  -> confirmed -> ask_email (play video8 once, prompt=email patterns)
  -> video8 ends -> recording_email (video1 loops, record 10s)
  -> confirmed -> confirm_all (show all data, confirm/cancel)
  -> confirmed -> submitting (POST /api/submit-patient)
  -> submitted -> farewell (play video4 once)
  -> video4 ends -> farewell_idle (video1 loops for 5s)
  -> 5s elapsed -> final (play video5 once)
  -> video5 ends -> idle (back to video1 loop)
```

Recording states: Show recording UI, call `recordAndTranscribe()`, show result with confirm/retry. On confirm, store data and advance. On retry, re-record.

The recording prompts (from controller.py):
- `recording_name`: no prompt
- `recording_question`: no prompt
- `recording_cnp`: `initial_prompt = '1 2 3 4 5 6 7 8 9 1 2 3 4'`
- `recording_email`: `initial_prompt = 'tudor.trocaru arond gmail punct com, radu.popescu arond yahoo punct com, ion.ionescu arond gmail punct com'`

Export these functions from workflow.ts:
- `initWorkflow()`: No-op for now, just sets initial state to stopped.
- `startWorkflow()`: Transition from stopped to idle (starts idle video loop).
- `stopWorkflow()`: Abort any active recording, clear data, transition to stopped, hide video and transcription panel.
- `getWorkflowState()`: Returns currentState.
- `getPatientData()`: Returns patientData (read-only copy).
- `onPersonEntered()`: If in idle state, transition to greeting. Otherwise ignore.

CRITICAL anti-patterns to avoid (from RESEARCH.md):
- Do NOT use `startInstructionSequence()` from video.ts. Use `playSingleVideo()` for each video individually.
- Do NOT share state via globals. Call functions explicitly.
- Do NOT use timers for video transitions. Use the `onEnded` callback from `playSingleVideo()`.
- Abort active recording on timeout/emergency stop via the `activeAbort` pattern from RESEARCH.md Pitfall 4.

For recording abort, keep a reference to a cancel function:
```typescript
let activeAbort: (() => void) | null = null;
```
On timeout or emergency stop, call `activeAbort()` before transitioning. This is conceptual -- since `recordAndTranscribe` returns a Promise and you cannot cancel a MediaRecorder from outside easily, the pragmatic approach is: set a `cancelled` flag, and when the recording Promise resolves, check the flag and discard results if cancelled. The recording will run to completion (10s max) but results are ignored.

For the `confirm_all` state (WKFL-05): call `showConfirmationSummary(patientData, onConfirm, onCancel)` from ui.ts. On confirm, transition to submitting. On cancel, transition to idle (clear data).

For the `submitting` state (WKFL-04): call `apiSubmitPatient(patientData)`, then transition to farewell on success.
